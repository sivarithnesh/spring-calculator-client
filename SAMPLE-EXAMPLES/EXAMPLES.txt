--1.qry_NewMsgFeedData
SELECT 
    TLMADM_MESSAGE_FEED.SUB_ACC_NO,
    TLMADM_BANK.LOCAL_ACC_NO,
    TLMADM_MESSAGE_FEED.DESCRIPTION AS "Statement Expected",
    TLMADM_MESSAGE_FEED.BSLA AS "Contact #",
    TLMADM_MESSAGE_FEED.BUS_CODE AS "Initial Method",
    TLMADM_MESSAGE_FEED.BUS_OWNER AS "TA Email ID",
    TLMADM_MESSAGE_FEED.STATUS
FROM TLMADM_BANK
INNER JOIN TLMADM_MESSAGE_FEED
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_MESSAGE_FEED.CORR_ACC_NO
WHERE 
    TLMADM_MESSAGE_FEED.DESCRIPTION IS NOT NULL
    AND TRIM(TLMADM_MESSAGE_FEED.DESCRIPTION) <> ''
    AND TLMADM_MESSAGE_FEED.STATUS = 0
    AND TLMADM_BANK.AGENT_CODE LIKE 'MUTUAL FUNDS%';
	
	--2.qry_TA_Names
	SELECT 
    TLMADM_ITEM.STRING_27 AS BBHNum,
    MAX(TLMADM_ITEM.STRING_5) AS "TA Name"
FROM TLMADM_BANK
INNER JOIN TLMADM_ITEM
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
WHERE 
    TLMADM_BANK.AGENT_CODE LIKE 'MUTUAL FUNDS%'
    AND TLMADM_ITEM.LS_TYPE = 9030
    AND TLMADM_ITEM.FLAG_2 = 0
GROUP BY 
    TLMADM_ITEM.STRING_27
HAVING 
    MAX(TLMADM_ITEM.STRING_5) NOT IN ('N', 'Y', '')
ORDER BY 
    MAX(TLMADM_ITEM.SYS_ENTRY_DATE) DESC;
	
	
--3.TBL_MASTERPOSITIONbREAKS 	

SELECT 
    TLMADM_BANK.AGENT_CODE,
    TLMADM_BANK.LOCAL_ACC_NO,

    CASE 
        WHEN TLMADM_STCK.STOCK_CODE IS NULL THEN 'N/A'
        ELSE TLMADM_STCK.STOCK_CODE
    END AS ISIN,

    TLMADM_ITEM.STRING_27,
    TLMADM_ITEM.STRING_28,

    CASE 
        WHEN TLMADM_ITEM.LS_TYPE = 9030 THEN TLMADM_ITEM.AMOUNT_6
        ELSE 0
    END AS BbhAmount,

    CASE 
        WHEN TLMADM_ITEM.LS_TYPE = 9030 THEN 0
        ELSE TLMADM_ITEM.AMOUNT_6
    END AS NsccAmount,

    TLMADM_ITEM.STMT_DATE,

    CASE 
        WHEN TLMADM_ITEM.STRING_14 IS NOT NULL 
             AND TRIM(TLMADM_ITEM.STRING_14) <> '' 
        THEN TLMADM_ITEM.STRING_14
        WHEN SUBSTR(TLMADM_BANK.LOCAL_ACC_NO, 8, 1) = '-' 
        THEN SUBSTR(TLMADM_BANK.LOCAL_ACC_NO, 1, 7)
        ELSE 'OMNIBUS'
    END AS BbhAcctNum,

    TLMADM_ITEM.STRING_21 AS BBHExtAcctNum,
    TLMADM_ITEM.STRING_21 AS NsccExtAcctNum,

    TLMADM_ITEM.AMOUNT_5,
    'System Entry Date - ' || TLMADM_ITEM.SYS_ENTRY_DATE AS HitsTlm,

    'NSCC' AS SourceDB,

    TLMADM_ITEM.STRING_29 AS BbhAcctName,
    TLMADM_ITEM.STRING_13

FROM TLMADM_BANK
INNER JOIN TLMADM_ITEM
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO

LEFT JOIN TLMADM_STCK
    ON TLMADM_ITEM.STOCK_ID = TLMADM_STCK.STOCK_ID

WHERE 
    TLMADM_BANK.AGENT_CODE = 'MUTUAL FUNDS - NSCC TRANS POS'
    AND TLMADM_ITEM.LS_TYPE IN (9030, 9099)
    AND TLMADM_ITEM.FLAG_2 = 0;
	

---4.TBL_BREAKSDETAILS_MONTHLYANDHEDGE

SELECT 
    TLMADM_BANK.LOCAL_ACC_NO,

    CASE 
        WHEN TLMADM_STCK.STOCK_CODE IS NULL THEN 'N/A'
        ELSE TLMADM_STCK.STOCK_CODE
    END AS ISIN,

    CASE 
        WHEN TLMADM_ITEM.STRING_14 IS NOT NULL 
             AND TRIM(TLMADM_ITEM.STRING_14) <> '' 
        THEN TLMADM_ITEM.STRING_14
        WHEN SUBSTR(TLMADM_BANK.LOCAL_ACC_NO, 8, 1) = '-' 
        THEN SUBSTR(TLMADM_BANK.LOCAL_ACC_NO, 1, 7)
        ELSE 'OMNIBUS'
    END AS BbhAcctNum,

    TLMADM_ITEM.STMT_DATE,

    CASE 
        WHEN TLMADM_ITEM.LS_TYPE IN (9106, 9129) 
        THEN TLMADM_ITEM.AMOUNT_6 * -1
        ELSE TLMADM_ITEM.AMOUNT_6
    END AS Expr1,

    TRUNC(TLMADM_ITEM.SYS_ENTRY_DATE) AS BreakOpened,

    TLMADM_ITEM.ITEM_ID AS BreakDesc,
    TLMADM_ITEM.STRING_6,

    TLMADM_DCOD.CODE_NAME,

    TLMPX_DBO_TLM_REF_REASON_CATEGORY.NAME AS ReasonCategory,
    'NSCC' AS SourceDB,

    TLMPX_DBO_TLM_REF_REASON_TYPE.NAME AS ReasonType,
    TLMPX_DBO_TLM_BDR_NOTE.NOTE,
    TLMADM_ITEM.ITEM_ID,

    CASE 
        WHEN TLMADM_ITEM.LS_TYPE IN (9105, 9106) 
        THEN NVL(TLMADM_ITEM.REFERENCE, '')
        ELSE ''
    END AS TAExtAcctNum,

    CASE 
        WHEN TLMADM_ITEM.LS_TYPE IN (9128, 9129) 
        THEN NVL(TLMADM_ITEM.STRING_21, '')
        ELSE ''
    END AS BBHExtAcctNum

FROM TLMADM_BANK

INNER JOIN TLMADM_ITEM
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO

LEFT JOIN TLMADM_DCOD
    ON TLMADM_ITEM.BRAN_CODE = TLMADM_DCOD.CODEC

LEFT JOIN TLMADM_STCK
    ON TLMADM_ITEM.STOCK_ID = TLMADM_STCK.STOCK_ID

LEFT JOIN TLMPX_DBO_TLM_REF_REASON_TYPE
    ON TLMADM_ITEM.REASON_TYPE_ID = TLMPX_DBO_TLM_REF_REASON_TYPE.REASON_TYPE_ID

LEFT JOIN TLMPX_DBO_TLM_REF_REASON_CATEGORY
    ON TLMPX_DBO_TLM_REF_REASON_TYPE.REASON_CATEGORY_ID = 
       TLMPX_DBO_TLM_REF_REASON_CATEGORY.REASON_CATEGORY_ID

LEFT JOIN TLMPX_DBO_TLM_BDR_NOTE
    ON TLMADM_ITEM.NOTE_ID = TLMPX_DBO_TLM_BDR_NOTE.ID

WHERE 
    TLMADM_BANK.AGENT_CODE = 'MUTUAL FUNDS - NSCC TRANS POS'
    AND TLMADM_ITEM.LS_TYPE NOT IN (9030, 9099)
    AND TLMADM_ITEM.FLAG_2 = 0;
	
--QRY_ESCAL_DETAILS_HEDGESETUP

SELECT 
    BD.Dept,
    BD.BreakType,

    MPB.Statement AS ParentBreakDate,
    MPB.Freq AS StatementDate,

    TRUNC(SYSDATE) - TRUNC(MPB.Statement) AS ParentAge,
    TRUNC(SYSDATE) - TRUNC(BD.BreakOpened) AS DiscvAge,

    MPB.SecID AS BBHNum,
    MPB.STOCK_CODE AS ISIN,
    MPB.SecName AS "Asset Description",

    MPB.BbhAcctNum,
    MPB.BbhAcctName,

    TO_CHAR(MPB.BbhAmount, 'FM9999999990.000000') AS "BBH Amount",
    TO_CHAR(MPB.NsccAmount, 'FM9999999990.000000') AS "Fund Amount",

    TO_CHAR(ABS(MPB.BbhAmount - MPB.NsccAmount), 
            'FM9999999990.000000') AS MasterBreak,

    BD.BreakQuantity AS ChildBreak,

    TO_CHAR(BD.BreakQuantity * BD.Price, 'FM9999999990.00') AS Exposure,

    BD.BreakDesc AS "Break Description",

    MPB.Comments AS ParentComments,
    BD.SourceDB,

    MPB.Freq,
    MPB.BBHExtAcctNum,
    MPB.NsccExtAcctNum AS FundExtAcctNum,

    TA."TA Name",

    MPB.BbhAcctNum || '-' || MPB.SecID AS SUB_ACC_NO

FROM tbl_MasterPositionBreaks MPB

INNER JOIN tbl_BreakDetails_MonthlyAndHedge BD
    ON MPB.LOCAL_ACC_NO = BD.LOCAL_ACC_NO
    AND MPB.BbhAcctNum = BD.BbhAcctNum
    AND MPB.Statement = BD.Statement
    AND MPB.STOCK_CODE = BD.ISIN

LEFT JOIN qry_TA_Names TA
    ON TA.BBHNum = MPB.SecID

WHERE 
    (TRUNC(SYSDATE) - TRUNC(BD.BreakOpened)) >= 30
    AND BD.SourceDB = 'Hedge'
    AND BD.ReasonType <> 'Resolved';

    ON Q.SUB_ACC_NO = MF.SUB_ACC_NO

ORDER BY 
    Q.Dept,
    Q.BreakType;
