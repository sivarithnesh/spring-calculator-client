/* ZeroPos_Monthly QUERY */

WITH ZeroPos_LedgersPriorMonth AS
(
    SELECT DISTINCT
        TLMADM_ITEM.STRING_27 AS SecID,
        TLMADM_ITEM.STRING_14 AS BbhAcctNum,
        TLMADM_ITEM.STMT_DATE AS Statement
    FROM TLMADM_BANK
    INNER JOIN TLMADM_ITEM
        ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
    WHERE TLMADM_ITEM.STMT_DATE >
          (TRUNC(SYSDATE) - EXTRACT(DAY FROM SYSDATE) - 4)
      AND (
            TLMADM_BANK.AGENT_CODE =
                'MUTUAL FUNDS - MONTHLY POSITIONS'
            OR TLMADM_BANK.AGENT_CODE LIKE
                'MUTUAL FUNDS - HEDGE%'
          )
      AND TLMADM_ITEM.LS_TYPE = 9030
),

ZeroPos_LedgersPriorMonth2 AS
(
    SELECT
        TLMADM_BANK.AGENT_CODE,
        TLMADM_ITEM.STRING_27 AS SecID,
        TLMADM_ITEM.STRING_28 AS SecName,
        TLMADM_ITEM.STMT_DATE AS Statement,
        TLMADM_ITEM.STRING_14 AS BbhAcctNum,
        TLMADM_ITEM.STRING_29 AS BbhAcctName,
        TLMADM_ITEM.STRING_21 AS BBHExtAcctNum,
        TLMADM_ITEM.STRING_20,
        TLMADM_ITEM.STRING_23 AS Reg,
        TLMADM_ITEM.STRING_24 AS TYPE,
        TLMADM_ITEM.STRING_30,
        TLMADM_ITEM.STRING_16 AS AssetClass,
        TLMADM_ITEM.STRING_17 AS "Seg/Omni",
        TLMADM_ITEM.STRING_12 AS ISIN,
        TLMADM_ITEM.STRING_13 AS Freq,
        TLMADM_ITEM.STRING_5 AS TA
    FROM TLMADM_BANK
    INNER JOIN TLMADM_ITEM
        ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
    WHERE (
            TLMADM_BANK.AGENT_CODE =
                'MUTUAL FUNDS - MONTHLY POSITIONS'
            OR TLMADM_BANK.AGENT_CODE LIKE
                'MUTUAL FUNDS - HEDGE%'
          )
      AND TLMADM_ITEM.STMT_DATE BETWEEN
            TRUNC(ADD_MONTHS(SYSDATE, -2), 'MM')
            AND LAST_DAY(ADD_MONTHS(SYSDATE, -1))
      AND TLMADM_ITEM.LS_TYPE = 9030
)

SELECT
    Z2.*
FROM ZeroPos_LedgersPriorMonth2 Z2
LEFT JOIN ZeroPos_LedgersPriorMonth Z1
    ON Z2.SecID = Z1.SecID
   AND Z2.BbhAcctNum = Z1.BbhAcctNum
WHERE Z1.SecID IS NULL
  AND Z2.Freq = 'MONTHLY'
ORDER BY
    Z2.AGENT_CODE,
    Z2.SecID,
    Z2.BbhAcctNum;

/* ZeroPos_Quarterly QUERY */

WITH ZeroPos_LedgersPriorMonth_Qrtly AS
(
    SELECT DISTINCT
        TLMADM_ITEM.STRING_27 AS SecID,
        TLMADM_ITEM.STRING_14 AS BbhAcctNum,
        TLMADM_ITEM.STMT_DATE AS Statement
    FROM TLMADM_BANK
    INNER JOIN TLMADM_ITEM
        ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
    WHERE TLMADM_ITEM.STMT_DATE >
          (TRUNC(SYSDATE) - EXTRACT(DAY FROM SYSDATE) - 4)
      AND TLMADM_BANK.AGENT_CODE LIKE
          'MUTUAL FUNDS - HEDGE%'
      AND TLMADM_ITEM.LS_TYPE = 9030
      AND TLMADM_ITEM.STRING_13 = 'QUARTERLY'
),

ZeroPos_LedgersPriorMonth2_Qrtly AS
(
    SELECT
        TLMADM_BANK.AGENT_CODE,
        TLMADM_ITEM.STRING_27 AS SecID,
        TLMADM_ITEM.STRING_28 AS SecName,
        TLMADM_ITEM.STMT_DATE AS Statement,
        TLMADM_ITEM.STRING_14 AS BbhAcctNum,
        TLMADM_ITEM.STRING_29 AS BbhAcctName,
        TLMADM_ITEM.STRING_21 AS BBHExtAcctNum,
        TLMADM_ITEM.STRING_20,
        TLMADM_ITEM.STRING_23 AS Reg,
        TLMADM_ITEM.STRING_24 AS TYPE,
        TLMADM_ITEM.STRING_30,
        TLMADM_ITEM.STRING_16 AS AssetClass,
        TLMADM_ITEM.STRING_17 AS "Seg/Omni",
        TLMADM_ITEM.STRING_12 AS ISIN,
        TLMADM_ITEM.STRING_13 AS Freq,
        TLMADM_ITEM.STRING_5 AS TA
    FROM TLMADM_BANK
    INNER JOIN TLMADM_ITEM
        ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
    WHERE TLMADM_BANK.AGENT_CODE LIKE
          'MUTUAL FUNDS - HEDGE%'
      AND TLMADM_ITEM.STMT_DATE BETWEEN
          (
              ADD_MONTHS(TRUNC(SYSDATE), -3)
              - EXTRACT(DAY FROM ADD_MONTHS(SYSDATE, -3))
              - 4
          )
          AND
          (
              ADD_MONTHS(TRUNC(SYSDATE), -3)
              - EXTRACT(DAY FROM ADD_MONTHS(SYSDATE, -3))
          )
      AND TLMADM_ITEM.STRING_13 = 'QUARTERLY'
      AND TLMADM_ITEM.LS_TYPE = 9030
)

SELECT
    Z2.*
FROM ZeroPos_LedgersPriorMonth2_Qrtly Z2
LEFT JOIN ZeroPos_LedgersPriorMonth_Qrtly Z1
    ON Z2.BbhAcctNum = Z1.BbhAcctNum
   AND Z2.SecID = Z1.SecID
WHERE Z1.SecID IS NULL
ORDER BY
    Z2.AGENT_CODE,
    Z2.SecID,
    Z2.BbhAcctNum;
