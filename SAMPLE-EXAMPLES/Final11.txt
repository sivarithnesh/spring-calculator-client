WITH qry_NewMsgFeedData AS (
    SELECT 
        MF.SUB_ACC_NO,
        B.LOCAL_ACC_NO,
        MF.DESCRIPTION AS Statement_Expected,
        MF.BSLA AS Contact_No,
        MF.BUS_CODE AS Initial_Method,
        B.BUS_OWNER AS TA_Email_ID,
        MF.STATUS
    FROM TLMADM_BANK B
    INNER JOIN TLMADM_MESSAGE_FEED MF 
        ON B.CORR_ACC_NO = MF.CORR_ACC_NO
    WHERE MF.DESCRIPTION <> ''
      AND MF.STATUS = 0
      AND B.AGENT_CODE LIKE 'MUTUAL FUNDS%'
),

qry_TA_Names AS (
    SELECT 
        I.STRING_27 AS BBHNum,
        MAX(I.STRING_5) AS TA_Name
    FROM TLMADM_BANK B
    INNER JOIN TLMADM_ITEM I 
        ON B.CORR_ACC_NO = I.CORR_ACC_NO
    WHERE B.AGENT_CODE LIKE 'MUTUAL FUNDS%'
      AND I.LS_TYPE = 9030
      AND I.FLAG_2 = 0
    GROUP BY I.STRING_27
),

tbl_MasterPositionBreaks AS (
    SELECT 
        B.AGENT_CODE,
        B.LOCAL_ACC_NO,
        NVL(ST.STOCK_CODE, 'N/A') AS ISIN,
        I.STRING_27,
        I.STRING_28,
        I.AMOUNT_6 AS NscAmount,
        I.STMT_DATE,
        I.STRING_21 AS BBHExtAcctNum,
        I.STRING_29 AS BbhAcctName
    FROM TLMADM_BANK B
    INNER JOIN TLMADM_ITEM I 
        ON B.CORR_ACC_NO = I.CORR_ACC_NO
    LEFT JOIN TLMADM_STCK ST 
        ON I.STOCK_ID = ST.STOCK_ID
    WHERE B.AGENT_CODE = 'MUTUAL FUNDS - NSCC TRANS POS'
      AND I.LS_TYPE IN (9030, 9099)
      AND I.FLAG_2 = 0
),

tbl_BreakDetails_MonthlyAndHedge AS (
    SELECT 
        B.LOCAL_ACC_NO,
        NVL(ST.STOCK_CODE, 'N/A') AS ISIN,
        I.STRING_14,
        I.STMT_DATE,
        I.AMOUNT_6,
        I.SYS_ENTRY_DATE AS BreakOpened,
        I.ITEM_DESC AS BreakDesc,
        'NSCC' AS SourceDB
    FROM TLMADM_BANK B
    INNER JOIN TLMADM_ITEM I 
        ON B.CORR_ACC_NO = I.CORR_ACC_NO
    LEFT JOIN TLMADM_STCK ST 
        ON I.STOCK_ID = ST.STOCK_ID
    WHERE B.AGENT_CODE = 'MUTUAL FUNDS - NSCC TRANS POS'
      AND I.LS_TYPE NOT IN (9030, 9099)
      AND I.FLAG_2 = 0
),

qry_Escal_Details_HedgeSetup AS (
    SELECT 
        BD.LOCAL_ACC_NO,
        BD.ISIN,
        MP.STRING_27 AS BBHNum,
        MP.BbhAcctName,
        BD.AMOUNT_6 AS BbhAmount,
        BD.BreakDesc,
        BD.SourceDB,
        TA.TA_Name
    FROM tbl_MasterPositionBreaks MP
    INNER JOIN tbl_BreakDetails_MonthlyAndHedge BD
        ON MP.LOCAL_ACC_NO = BD.LOCAL_ACC_NO
       AND MP.ISIN = BD.ISIN
    LEFT JOIN qry_TA_Names TA
        ON TA.BBHNum = MP.STRING_27
)

-- FINAL MAIN QUERY
SELECT 
    E.*,
    N.Initial_Method,
    N.TA_Email_ID,
    N.Statement_Expected,
    N.Contact_No
FROM qry_Escal_Details_HedgeSetup E
LEFT JOIN qry_NewMsgFeedData N
    ON E.LOCAL_ACC_NO = N.SUB_ACC_NO
ORDER BY E.LOCAL_ACC_NO;

WITH qry_NewMsgFeedData AS (
SELECT 
    TLMADM_MESSAGE_FEED.SUB_ACC_NO,
    TLMADM_BANK.LOCAL_ACC_NO,
    TLMADM_MESSAGE_FEED.DESCRIPTION AS "Statement Expected",
    TLMADM_MESSAGE_FEED.BSLA AS "Contact #",
    TLMADM_MESSAGE_FEED.BUS_CODE AS "Initial Method",
    p.BUS_OWNER AS "TA Email ID",
    TLMADM_MESSAGE_FEED.STATUS
FROM TLMADM_BANK 
INNER JOIN TLMADM_MESSAGE_FEED 
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_MESSAGE_FEED.CORR_ACC_NO
WHERE TLMADM_MESSAGE_FEED.DESCRIPTION <> ''
  AND TLMADM_MESSAGE_FEED.STATUS = 0
  AND TLMADM_BANK.AGENT_CODE LIKE 'MUTUAL FUNDS%'
),

qry_TA_Names AS (
SELECT 
    TLMADM_ITEM.STRING_27 AS BBHNum,
    MIN(TLMADM_ITEM.STRING_5) AS "TA Name"
FROM TLMADM_BANK 
INNER JOIN TLMADM_ITEM 
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
WHERE TLMADM_BANK.AGENT_CODE LIKE 'MUTUAL FUNDS%'
  AND TLMADM_ITEM.LS_TYPE = 9030 
  AND TLMADM_ITEM.FLAG_2 = 0
GROUP BY TLMADM_ITEM.STRING_27
),

tbl_MasterPositionBreaks AS (
SELECT 
    TLMADM_BANK.AGENT_CODE,
    TLMADM_BANK.LOCAL_ACC_NO,
    CASE WHEN TLMADM_STCK.STOCK_CODE IS NULL THEN 'N/A' ELSE TLMADM_STCK.STOCK_CODE END AS ISIN,
    TLMADM_ITEM.STRING_27,
    TLMADM_ITEM.STRING_28,
    CASE WHEN TLMADM_ITEM.LS_TYPE = 9030 THEN TLMADM_ITEM.AMOUNT_6 ELSE 0 END AS BbhAmount,
    CASE WHEN TLMADM_ITEM.LS_TYPE = 9030 THEN TLMADM_ITEM.AMOUNT_6 ELSE 0 END AS NscAmount,
    TLMADM_ITEM.STMT_DATE,
    CASE WHEN TLMADM_ITEM.STRING_14 IS NULL THEN TLMADM_ITEM.STRING_21 ELSE TLMADM_ITEM.STRING_21 END AS BBHExtAcctNum,
    TLMADM_ITEM.STRING_29 AS BbhAcctName,
    CASE WHEN TLMADM_ITEM.SYS_ENTRY_DATE IS NOT NULL THEN 'System Entry Date = ' || TLMADM_ITEM.SYS_ENTRY_DATE END AS Comments,
    'NSCC' AS SourceDB
FROM TLMADM_BANK 
INNER JOIN TLMADM_ITEM 
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
LEFT JOIN TLMADM_STCK 
    ON TLMADM_ITEM.STOCK_ID = TLMADM_STCK.STOCK_ID
WHERE TLMADM_BANK.AGENT_CODE = 'MUTUAL FUNDS - NSCC TRANS POS'
  AND TLMADM_ITEM.LS_TYPE IN (9030, 9099)
  AND TLMADM_ITEM.FLAG_2 = 0
),

tbl_BreakDetails_MonthlyAndHedge AS (
SELECT 
    TLMADM_BANK.LOCAL_ACC_NO,
    CASE WHEN TLMADM_STCK.STOCK_CODE IS NULL THEN 'N/A' ELSE TLMADM_STCK.STOCK_CODE END AS ISIN,
    CASE WHEN TLMADM_ITEM.STRING_14 <> '' THEN TLMADM_ITEM.STRING_14 ELSE NULL END,
    CASE 
        WHEN SUBSTR(TLMADM_BANK.LOCAL_ACC_NO,8,1) = '-' 
        THEN SUBSTR(TLMADM_BANK.LOCAL_ACC_NO,1,7) 
        ELSE 'OMNIBUS' 
    END AS BbhAcctNum,
    CASE 
        WHEN TLMADM_ITEM.LS_TYPE IN (9106,9129) 
        THEN TLMADM_ITEM.AMOUNT_6 * -1 
        ELSE TLMADM_ITEM.AMOUNT_6 
    END AS Expr1,
    TRUNC(TLMADM_ITEM.SYS_ENTRY_DATE) AS BreakOpened,
    TLMADM_ITEM.ITEM_DESC AS BreakDesc,
    TLMADM_ITEM.STRING_6,
    TLMADM_DCOD.CODE_NAME,
    TLMPX_DBO_TLM_REF_REASON_CATEGORY.NAME AS ReasonCategory,
    'NSCC' AS SourceDB,
    TLMPX_DBO_TLM_REF_REASON_TYPE.NAME AS ReasonType,
    TLMPX_DBO_TLM_BDR_NOTE.NOTE,
    TLMADM_ITEM.ITEM_ID,
    CASE WHEN TLMADM_ITEM.LS_TYPE IN (9105,9106) THEN TLMADM_ITEM.REFERENCE ELSE '' END AS TAExtAcctNum,
    CASE WHEN TLMADM_ITEM.LS_TYPE IN (9128,9129) THEN TLMADM_ITEM.STRING_21 ELSE '' END AS BBHExtAcctNum
FROM TLMADM_BANK 
INNER JOIN TLMADM_ITEM 
    ON TLMADM_BANK.CORR_ACC_NO = TLMADM_ITEM.CORR_ACC_NO
LEFT JOIN TLMADM_DCOD 
    ON TLMADM_ITEM.BRAN_CODE = TLMADM_DCOD.CODE
LEFT JOIN TLMADM_STCK 
    ON TLMADM_ITEM.STOCK_ID = TLMADM_STCK.STOCK_ID
LEFT JOIN TLMPX_DBO_TLM_REF_REASON_TYPE 
    ON TLMADM_ITEM.REASON_TYPE_ID = TLMPX_DBO_TLM_REF_REASON_TYPE.REASON_TYPE_ID
LEFT JOIN TLMPX_DBO_TLM_REF_REASON_CATEGORY 
    ON TLMPX_DBO_TLM_REF_REASON_TYPE.REASON_CATEGORY_ID = TLMPX_DBO_TLM_REF_REASON_CATEGORY.REASON_CATEGORY_ID
LEFT JOIN TLMPX_DBO_TLM_BDR_NOTE 
    ON TLMADM_ITEM.ITEM_ID = TLMPX_DBO_TLM_BDR_NOTE.ID
WHERE TLMADM_BANK.AGENT_CODE = 'MUTUAL FUNDS - NSCC TRANS POS'
  AND TLMADM_ITEM.LS_TYPE NOT IN (9030,9099)
  AND TLMADM_ITEM.FLAG_2 = 0
),

qry_Escal_Details_HedgeSetup AS (
SELECT 
    tbl_BreakDetails_MonthlyAndHedge.LOCAL_ACC_NO,
    tbl_BreakDetails_MonthlyAndHedge.ISIN,
    tbl_MasterPositionBreaks.STRING_27 AS BBHNum,
    tbl_MasterPositionBreaks.BbhAcctName,
    tbl_MasterPositionBreaks.BbhAmount,
    tbl_BreakDetails_MonthlyAndHedge.BreakDesc,
    tbl_BreakDetails_MonthlyAndHedge.SourceDB,
    qry_TA_Names."TA Name"
FROM tbl_MasterPositionBreaks
INNER JOIN tbl_BreakDetails_MonthlyAndHedge
    ON tbl_MasterPositionBreaks.LOCAL_ACC_NO = tbl_BreakDetails_MonthlyAndHedge.LOCAL_ACC_NO
   AND tbl_MasterPositionBreaks.ISIN = tbl_BreakDetails_MonthlyAndHedge.ISIN
LEFT JOIN qry_TA_Names
    ON qry_TA_Names.BBHNum = tbl_MasterPositionBreaks.STRING_27
)

SELECT 
    qry_Escal_Details_HedgeSetup.*,
    qry_NewMsgFeedData."Initial Method",
    qry_NewMsgFeedData."TA Email ID",
    qry_NewMsgFeedData."Statement Expected",
    qry_NewMsgFeedData."Contact #"
FROM qry_Escal_Details_HedgeSetup
LEFT JOIN qry_NewMsgFeedData
    ON qry_Escal_Details_HedgeSetup.LOCAL_ACC_NO = qry_NewMsgFeedData.SUB_ACC_NO
ORDER BY qry_Escal_Details_HedgeSetup.LOCAL_ACC_NO;
